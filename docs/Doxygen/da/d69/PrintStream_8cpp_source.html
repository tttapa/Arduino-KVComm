189 template <class T>
190 void printHex(Print &printer, T val) {
191  if (showbasePrintStream)
192  printer.print("0x");
193  bool nonZero = false;
194  for (int i = sizeof(val) - 1; i >= 0; i--) {
195  uint8_t currByte = ((uint8_t *)&val)[i];
196  if (currByte != 0 || i == 0)
197  nonZero = true;
198  if (leadingZerosPrintStream || nonZero) {
199  printer.print(nibble_to_hex(currByte >> 4));
200  printer.print(nibble_to_hex(currByte));
201  if (byteSeparatorPrintStream && i)
202  printer.print(byteSeparatorPrintStream);
203  }
204  }
205 }

207 template <class T>
208 void printBin(Print &printer, T val) {
209  if (showbasePrintStream)
210  printer.print("0b");
211  bool nonZero = false;
212  for (int i = sizeof(val) - 1; i >= 0; i--) {
213  uint8_t currByte = ((uint8_t *)&val)[i];
214  for (int j = 7; j >= 0; j--) {
215  uint8_t currBit = currByte & 0x80;
216  if (currBit != 0 || (i == 0 && j == 0))
217  nonZero = true;
218  if (leadingZerosPrintStream || nonZero)
219  printer.print(currBit ? '1' : '0');
220  currByte <<= 1;
221  }
222  if (byteSeparatorPrintStream && i &&
223  (leadingZerosPrintStream || nonZero))
224  printer.print(byteSeparatorPrintStream);
225  }
226 }
